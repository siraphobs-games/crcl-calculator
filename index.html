  <script>

    function calculateCrCl() {

      const sex = document.getElementById("sex").value;

      const age = parseFloat(document.getElementById("age").value);

      const weight = parseFloat(document.getElementById("weight").value);

      const scr = parseFloat(document.getElementById("scr").value);


      if (!age || !weight || !scr || scr <= 0) {

        document.getElementById("result").innerText = "กรุณากรอกข้อมูลให้ถูกต้อง";

        return;

      }


      let crcl = ((140 - age) * weight) / (72 * scr);

      if (sex === "female") crcl *= 0.85;


      document.getElementById("result").innerText = `CrCl ≈ ${crcl.toFixed(2)} mL/min`;

    }

  </script>
